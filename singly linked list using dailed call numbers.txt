#include <stdio.h>
#include <stdlib.h>
struct contact{
    long int number;
    struct contact *next;
};
int main(){
    struct contact *ct1 = (struct contact*)malloc(sizeof(struct contact));
    struct contact *ct2 = (struct contact*)malloc(sizeof(struct contact));
    struct contact *ct3 = (struct contact*)malloc(sizeof(struct contact));
    ct1->number = 101;
    ct2->number = 102;
    ct3->number = 103;
    ct1->next = ct2;
    ct2->next = ct3;
    ct3->next = NULL;
    struct contact *head = ct1;
    while(head!=NULL){
        printf("%d---->",head->number);
        head = head->next;
    }
    printf("NULL");
    return 0;
}
