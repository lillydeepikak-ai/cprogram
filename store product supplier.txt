#include <stdio.h>
#include <stdlib.h>
struct Supplier {
    char supplierName[50];
    char supplierLocation[50];
};
struct Product {
    int productID;
    char productName[50];
    float productPrice;
    struct Supplier supplier;
};

int main() {
    struct Product *ptr;
    ptr = (struct Product *)malloc(sizeof(struct Product));

    if (ptr == NULL) {
        printf("Memory allocation failed!\n");
        return 1;
    }
    printf("Enter Product ID: ");
    scanf("%d", &ptr->productID);
    printf("Enter Product Name: ");
    scanf(" %[^\n]", ptr->productName);
    printf("Enter Product Price: ");
    scanf("%f", &ptr->productPrice);
    printf("Enter Supplier Name: ");
    scanf(" %[^\n]", ptr->supplier.supplierName);
    printf("Enter Supplier Location: ");
    scanf(" %[^\n]", ptr->supplier.supplierLocation);
    printf("\n--- Product Details ---\n");
    printf("Product ID    : %d\n", ptr->productID);
    printf("Product Name  : %s\n", ptr->productName);
    printf("Product Price : %.2f\n", ptr->productPrice);
    printf("Supplier Name : %s\n", ptr->supplier.supplierName);
    printf("Supplier Location: %s\n", ptr->supplier.supplierLocation);
   
    free(ptr);

    return 0;
}
